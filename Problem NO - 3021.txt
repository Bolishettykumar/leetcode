class Solution {
    public long flowerGame(int n, int m) {
        long xEven = n / 2;         // count of even numbers in [1..n]
        long xOdd = (n + 1) / 2;    // count of odd numbers in [1..n]
        long yEven = m / 2;         // count of even numbers in [1..m]
        long yOdd = (m + 1) / 2;    // count of odd numbers in [1..m]
        
        // Alice wins when x + y is odd
        return xEven * yOdd + xOdd * yEven;
    }
}
